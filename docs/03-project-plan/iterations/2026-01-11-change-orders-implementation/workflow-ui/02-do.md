### Bug Fix: 404 Error When Updating Change Orders

**Issue:** Users received 404 "not found" errors when updating Change Order status from "Draft" to "Submitted for Approval".

**Root Cause:** PostgreSQL's `CURRENT_TIMESTAMP` (also called `now()`) returns the **transaction start time**, not the actual current time. When tests run within a transaction (as the `db_session` fixture does), the following sequence occurred:

1. Transaction starts → `CURRENT_TIMESTAMP` = T1
2. `CreateVersionCommand` sets `valid_time` to `[datetime.now(UTC), NULL)` = [T2, NULL) where T2 > T1
3. Query uses `.op("@>")(func.current_timestamp())` which evaluates to T1 (transaction start time)
4. Range [T2, NULL) does NOT contain T1 because T1 < T2
5. Query returns no results → 404 error

**Solution:** Changed queries to use `func.clock_timestamp()` instead of `func.current_timestamp()`. The `clock_timestamp()` function returns the actual current time, not the transaction start time.

```python
"""Tests for ChangeOrderService workflow metadata in public schema."""

from unittest.mock import AsyncMock, MagicMock
from uuid import uuid4

import pytest
from sqlalchemy.ext.asyncio import AsyncSession

from app.models.domain.branch import Branch
from app.models.domain.change_order import ChangeOrder
from app.models.schemas.change_order import ChangeOrderPublic
from app.services.branch_service import BranchService
from app.services.change_order_service import ChangeOrderService
from app.services.change_order_workflow_service import ChangeOrderWorkflowService


@pytest.mark.asyncio
async def test_to_public_includes_available_transitions(db_session: AsyncSession):
    """Verify that _to_public() adds available_transitions field from workflow service.

    Acceptance Criteria:
    - Backend ChangeOrderPublic schema includes available_transitions field
    - Field is populated by calling workflow_service.get_available_transitions()

    This test ensures the frontend can receive valid workflow transitions
    to dynamically filter the status dropdown.
    """
    # Arrange
    co = ChangeOrder(
        id=uuid4(),
        change_order_id=uuid4(),
        code="CO-001",
        project_id=uuid4(),
        title="Test Change Order",
        status="Draft",
        branch="main",
        created_by=uuid4(),
        created_at=None,  # Will be set by transaction
        updated_by=None,
        updated_at=None,
        parent_id=None,
        deleted_at=None,
    )

    # Create service with mocked workflow service
    service = ChangeOrderService(db_session)

    # Mock workflow service to return specific transitions
    service.workflow = MagicMock(spec=ChangeOrderWorkflowService)
    service.workflow.get_available_transitions = MagicMock(
        return_value=["Submitted for Approval"]
    )

    # Mock branch service for branch_locked check
    service.branch_service = AsyncMock(spec=BranchService)

    # Act
    result = service._to_public(co)

    # Assert
    assert isinstance(result, ChangeOrderPublic), "Result should be ChangeOrderPublic schema"
    assert result.available_transitions == ["Submitted for Approval"], (
        "available_transitions should match workflow service response"
    )
    service.workflow.get_available_transitions.assert_called_once_with("Draft"), (
        "Workflow service should be called with current status"
    )
```
